  The existing `query(sql)` method at line 44 stays unchanged.  Add `query_typed` immediately after it:

  ```python
  # Column name used in canonical time-series Parquet files for the
  # observation concept / metric name.
  _CONCEPT_COLUMN = "concept"

  def query_typed(
      self,
      relative_path: str,
      *,
      concept: str | None = None,
      entity: str | None = None,
      start: str | None = None,
      end: str | None = None,
  ) -> pl.DataFrame:
      """Query a canonical Parquet file with typed filters.

      All filter parameters are optional; omitting them returns all rows.

      Args:
          relative_path: Parquet path relative to the canonical root,
              as passed to :meth:`write_parquet` / :meth:`upsert`.
          concept: Filter to rows where the ``concept`` column equals
              this value (e.g. ``"bank_rate"``).
          entity: Filter to rows where ``entity_id`` equals this value
              (e.g. ``"BoE"``).
          start: ISO-8601 date string (``"YYYY-MM-DD"``).  Rows where
              ``timestamp >= start`` are retained.
          end: ISO-8601 date string.  Rows where ``timestamp <= end``
              are retained.

      Returns:
          Filtered Polars DataFrame.

      Raises:
          FileNotFoundError: If *relative_path* does not exist.
          ValueError: If *start* or *end* are not valid ISO-8601 dates.
      """
      path = self._canonical_path(relative_path)
      if not path.exists():
          raise FileNotFoundError(f"Canonical file not found: {path}")

      # Validate date strings eagerly so the caller gets a clear error
      # rather than a cryptic DuckDB exception.
      from datetime import date as _date
      def _parse_date(s: str, param: str) -> str:
          try:
              _date.fromisoformat(s)
          except ValueError as exc:
              raise ValueError(
                  f"query_typed: {param}={s!r} is not a valid ISO-8601 date"
              ) from exc
          return s

      if start is not None:
          start = _parse_date(start, "start")
      if end is not None:
          end = _parse_date(end, "end")

      frame = pl.scan_parquet(path)
      schema = frame.collect_schema()

      if concept is not None and _CONCEPT_COLUMN in schema:
          frame = frame.filter(pl.col(_CONCEPT_COLUMN) == concept)

      if entity is not None and "entity_id" in schema:
          frame = frame.filter(pl.col("entity_id") == entity)

      if "timestamp" in schema:
          if start is not None:
              frame = frame.filter(pl.col("timestamp") >= pl.lit(start).str.to_date())
          if end is not None:
              frame = frame.filter(pl.col("timestamp") <= pl.lit(end).str.to_date())

      return frame.collect()
  ```

  **Design notes**:
  - Uses Polars lazy scanning (`scan_parquet`) so filters are pushed down before materialisation.
  - Column existence is checked against the schema before filtering; missing columns are silently skipped (tolerant of frames that pre-date the typed schema).
  - `concept` maps to the `concept` column; `entity` maps to `entity_id`.  This mapping is the authoritative contract — update `_CONCEPT_COLUMN` if the canonical column name changes.

- [ ] **Add typed-query tests to `packages/uk-data/tests/test_storage.py`**

  Append to the existing file (created in 02-02):

  ```python
  # ── typed query tests ───────────────────────────────────────────────────────

  def _make_typed_df() -> pl.DataFrame:
      """5-row canonical frame with concept + entity_id + timestamp + value."""
      from datetime import date
      return pl.DataFrame({
          "source":    ["boe", "boe", "ons", "ons", "ons"],
          "entity_id": ["BoE", "BoE", "ONS", "ONS", "ONS"],
          "concept":   ["bank_rate", "bank_rate", "gdp", "gdp", "cpi"],
          "timestamp": [
              date(2020, 1, 1),
              date(2021, 1, 1),
              date(2020, 6, 1),
              date(2022, 6, 1),
              date(2021, 3, 1),
          ],
          "value": [0.1, 0.25, 1.5, 2.0, 3.5],
      })


  def test_query_typed_no_filters(tmp_path):
      store = _make_store(tmp_path)
      df = _make_typed_df()
      store.write_parquet(df, "multi.parquet")
      result = store.query_typed("multi.parquet")
      assert len(result) == 5


  def test_query_typed_by_concept(tmp_path):
      store = _make_store(tmp_path)
      store.write_parquet(_make_typed_df(), "multi.parquet")
      result = store.query_typed("multi.parquet", concept="bank_rate")
      assert len(result) == 2
      assert set(result["concept"].to_list()) == {"bank_rate"}


  def test_query_typed_by_entity(tmp_path):
      store = _make_store(tmp_path)
      store.write_parquet(_make_typed_df(), "multi.parquet")
      result = store.query_typed("multi.parquet", entity="ONS")
      assert len(result) == 3
      assert all(e == "ONS" for e in result["entity_id"].to_list())


  def test_query_typed_date_range(tmp_path):
      store = _make_store(tmp_path)
      store.write_parquet(_make_typed_df(), "multi.parquet")
      result = store.query_typed("multi.parquet", start="2021-01-01", end="2021-12-31")
      assert all(
          "2021-01-01" <= str(ts) <= "2021-12-31"
          for ts in result["timestamp"].to_list()
      )


  def test_query_typed_combined_filters(tmp_path):
      store = _make_store(tmp_path)
      store.write_parquet(_make_typed_df(), "multi.parquet")
      result = store.query_typed(
          "multi.parquet",
          concept="gdp",
          entity="ONS",
          start="2022-01-01",
      )
      assert len(result) == 1
      assert result["value"][0] == 2.0


  def test_query_typed_raises_on_missing_file(tmp_path):
      store = _make_store(tmp_path)
      with pytest.raises(FileNotFoundError):
          store.query_typed("nonexistent.parquet")


  def test_query_typed_raises_on_bad_date(tmp_path):
      store = _make_store(tmp_path)
      store.write_parquet(_make_typed_df(), "multi.parquet")
      with pytest.raises(ValueError, match="ISO-8601"):
          store.query_typed("multi.parquet", start="not-a-date")


  def test_query_typed_tolerates_missing_concept_column(tmp_path):
      """Frame without a 'concept' column: concept filter is silently skipped."""
      store = _make_store(tmp_path)
      df = pl.DataFrame({
          "source": ["boe"],
          "entity_id": ["BoE"],
          "timestamp": [__import__('datetime').date(2024, 1, 1)],
          "value": [5.0],
      })
      store.write_parquet(df, "no_concept.parquet")
      # concept filter should be ignored, not raise
      result = store.query_typed("no_concept.parquet", concept="bank_rate")
      assert len(result) == 1
  ```
