# Plan 02-02: Upsert Semantics

## Goal
Add `CanonicalStore.upsert(df)` that merges a Polars DataFrame into the existing Parquet file using DuckDB `INSERT OR REPLACE` on the composite key `(source, entity_id, timestamp)`.

  ```python
  # Expected schema columns present in every canonical time-series frame:
  _UPSERT_PK = ("source", "entity_id", "timestamp")

  def upsert(self, df: pl.DataFrame, relative_path: str) -> Path:
      """Merge *df* into the Parquet file at *relative_path*.

      Rows with duplicate ``(source, entity_id, timestamp)`` keys replace
      existing rows; all other existing rows are retained.

      The Parquet file is created if it does not yet exist.

      Args:
          df: DataFrame to merge in.  Must contain at minimum the columns
              ``source`` (Utf8), ``entity_id`` (Utf8), and ``timestamp``
              (Datetime or Date).
          relative_path: Same relative path used with :meth:`write_parquet`.

      Returns:
          Path to the written Parquet file.

      Raises:
          ValueError: If *df* is missing any of the required PK columns.
      """
      missing = [col for col in _UPSERT_PK if col not in df.columns]
      if missing:
          raise ValueError(
              f"upsert() requires columns {_UPSERT_PK}; missing: {missing}"
          )

      path = self._canonical_path(relative_path)

      with duckdb.connect() as conn:
          # Register incoming data
          conn.register("incoming", df.to_arrow())

          if path.exists():
              conn.execute(
                  f"CREATE TABLE store AS SELECT * FROM read_parquet('{path}')"
              )
          else:
              # Bootstrap empty table with same schema as incoming
              conn.execute("CREATE TABLE store AS SELECT * FROM incoming WHERE false")

          # INSERT OR REPLACE semantics via DELETE + INSERT
          conn.execute(
              """
              DELETE FROM store
              WHERE (source, entity_id, timestamp) IN (
                  SELECT source, entity_id, timestamp FROM incoming
              )
              """
          )
          conn.execute("INSERT INTO store SELECT * FROM incoming")

          result = conn.execute("SELECT * FROM store").pl()

      path.parent.mkdir(parents=True, exist_ok=True)
      result.write_parquet(path)
      return path
  ```

  **Note on DuckDB version compatibility**: `duckdb.connect()` (no path) creates an in-memory database. Use `conn.execute("SELECT * FROM store").pl()` which returns a Polars DataFrame directly (DuckDB ≥ 0.9).  If `.pl()` is unavailable, fall back to `pl.from_arrow(conn.execute(...).arrow())`.

- [ ] **Update `packages/uk-data/tests/test_storage.py`** (create the file if absent)

  ```python
  # test_storage.py — required test cases

  import polars as pl
  import pytest
  from pathlib import Path
  from uk_data.storage.canonical import CanonicalStore


  def _make_store(tmp_path: Path) -> CanonicalStore:
      return CanonicalStore(tmp_path / "store")


  def _sample_df(source: str = "boe", rows: int = 3) -> pl.DataFrame:
      from datetime import date
      return pl.DataFrame({
          "source": [source] * rows,
          "entity_id": [f"entity_{i}" for i in range(rows)],
          "timestamp": [date(2024, 1, i + 1) for i in range(rows)],
          "value": [float(i) for i in range(rows)],
      })


  def test_upsert_creates_file_when_absent(tmp_path):
      store = _make_store(tmp_path)
      df = _sample_df()
      path = store.upsert(df, "boe/rates.parquet")
      assert path.exists()
      result = store.read_parquet("boe/rates.parquet")
      assert len(result) == 3


  def test_upsert_replaces_duplicate_rows(tmp_path):
      store = _make_store(tmp_path)
      df_initial = _sample_df()
      store.upsert(df_initial, "boe/rates.parquet")

      # Same keys, updated value
      from datetime import date
      df_update = pl.DataFrame({
          "source": ["boe"],
          "entity_id": ["entity_0"],
          "timestamp": [date(2024, 1, 1)],
          "value": [99.0],
      })
      store.upsert(df_update, "boe/rates.parquet")

      result = store.read_parquet("boe/rates.parquet")
      assert len(result) == 3  # row count unchanged
      updated = result.filter(
          (pl.col("entity_id") == "entity_0") &
          (pl.col("source") == "boe")
      )
      assert updated["value"][0] == 99.0


  def test_upsert_appends_new_rows(tmp_path):
      store = _make_store(tmp_path)
      store.upsert(_sample_df(rows=2), "boe/rates.parquet")

      from datetime import date
      df_new = pl.DataFrame({
          "source": ["boe"],
          "entity_id": ["entity_99"],
          "timestamp": [date(2024, 6, 1)],
          "value": [5.0],
      })
      store.upsert(df_new, "boe/rates.parquet")

      result = store.read_parquet("boe/rates.parquet")
      assert len(result) == 3  # 2 original + 1 new


  def test_upsert_raises_on_missing_pk_column(tmp_path):
      store = _make_store(tmp_path)
      bad_df = pl.DataFrame({"value": [1.0]})
      with pytest.raises(ValueError, match="missing"):
          store.upsert(bad_df, "boe/rates.parquet")


  def test_upsert_idempotent(tmp_path):
      """Upserting the same data twice yields the same row count."""
      store = _make_store(tmp_path)
      df = _sample_df()
      store.upsert(df, "boe/rates.parquet")
      store.upsert(df, "boe/rates.parquet")
      result = store.read_parquet("boe/rates.parquet")
      assert len(result) == 3
  ```
